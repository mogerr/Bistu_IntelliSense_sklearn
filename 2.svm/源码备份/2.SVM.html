# 模型构建
svm = SVC(kernel='linear', C=1.0, random_state=42, probability=True) # 创建 SVM 分类器，设置 probability 参数为 True，以便输出预测概率
# 使用训练集拟合模型
svm = svm.fit(X_train, y_train)

y_scores = svm.decision_function(X_test)
